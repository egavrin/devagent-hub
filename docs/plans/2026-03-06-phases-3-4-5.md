# DevAgent Hub Phases 3-4-5: End-to-End Workflow Implementation Plan

> **For Claude:** REQUIRED SUB-SKILL: Use superpowers:executing-plans to implement this plan task-by-task.

**Goal:** Wire triage, planning, implementation, verification, PR creation, auto-review, and repair into a complete end-to-end workflow driven by `devagent-hub run <issue>`.

**Architecture:** A `WorkflowOrchestrator` class coordinates the full lifecycle. It uses the existing `StateStore`, `GitHubGateway`, `WorktreeManager`, and `RunLauncher` modules. Each phase invokes `devagent workflow run` as a subprocess, collects structured `PhaseResult` output, posts summaries to GitHub, and transitions the state machine. The orchestrator is testable via `MockGitHubGateway` and a `MockRunLauncher`.

**Tech Stack:** Bun 1.3.9, TypeScript 5.7, Vitest, bun:sqlite, `gh` CLI.

```typescript
    open_requires: string[];
  };
  repair: {
    max_rounds: number;
  };
  handoff: {
    when: string[];
  };
}

const DEFAULTS: WorkflowConfig = {
  version: 1,
  tracker: { kind: "github", issue_labels_include: ["da:ready"] },
  dispatch: { max_concurrency: 1 },
  workspace: { mode: "worktree", root: ".devagent/workspaces" },
  runner: { approval_mode: "auto-edit", max_iterations: 30 },
  roles: { triage: "architect", plan: "architect", implement: "general", review: "reviewer" },
  verify: { commands: [] },
  pr: { draft: true, open_requires: [] },
  repair: { max_rounds: 2 },
  handoff: { when: [] },
};

export function parseWorkflowConfig(content: string): WorkflowConfig {
  // Extract YAML frontmatter between --- markers
  const match = content.match(/^---\n([\s\S]*?)\n---/);
  if (!match) return { ...DEFAULTS };

  // Parse YAML manually (simple key-value/list format) or use a library
  // For robustness, use a simple YAML parser
  const yaml = parseSimpleYaml(match[1]);
  return deepMerge(DEFAULTS, yaml) as WorkflowConfig;
}

export function loadWorkflowConfig(repoRoot: string): WorkflowConfig {
  const path = join(repoRoot, "WORKFLOW.md");
  if (!existsSync(path)) return { ...DEFAULTS };
  return parseWorkflowConfig(readFileSync(path, "utf-8"));
}
```

```typescript
  responses: Map<string, MockPhaseResponse> = new Map();
  launches: Array<{ phase: string; runId: string; input: unknown; repoPath: string }> = [];

    this.responses.set(phase, response);

    phase: string;
    repoPath: string;
    runId: string;
    input: unknown;
  }): LaunchResult {
    this.launches.push(params);

    const response = this.responses.get(params.phase) ?? {
      exitCode: 0,
      output: { schemaVersion: 1, phase: params.phase, result: {}, summary: "Mock result" },
    };

    return {
      exitCode: response.exitCode,
      outputPath: `/tmp/mock/${params.runId}/${params.phase}-output.json`,
      eventsPath: `/tmp/mock/${params.runId}/${params.phase}-events.jsonl`,
      output: response.output,
    };
  }
}
```

```typescript
import type { StateStore } from "../state/store.js";
import type { GitHubGateway } from "../github/gateway.js";
import type { MockRunLauncher } from "../runner/mock-launcher.js";
import type { RunLauncher, LaunchResult } from "../runner/launcher.js";
import type { WorkflowRun } from "../state/types.js";
import type { WorkflowConfig } from "./config.js";
import { loadWorkflowConfig } from "./config.js";

export interface OrchestratorDeps {
  store: StateStore;
  github: GitHubGateway;
  launcher: RunLauncher | MockRunLauncher;
  repo: string;
  repoRoot?: string;
  config?: WorkflowConfig;
}

export class WorkflowOrchestrator {
  private store: StateStore;
  private github: GitHubGateway;
  private launcher: RunLauncher | MockRunLauncher;
  private repo: string;
  private repoRoot: string;
  private config: WorkflowConfig;

  constructor(deps: OrchestratorDeps) {
    this.store = deps.store;
    this.github = deps.github;
    this.launcher = deps.launcher;
    this.repo = deps.repo;
    this.repoRoot = deps.repoRoot ?? process.cwd();
    this.config = deps.config ?? loadWorkflowConfig(this.repoRoot);
  }

  async triage(issueNumber: number): Promise<WorkflowRun> {
    // Fetch issue
    const issue = await this.github.fetchIssue(this.repo, issueNumber);

    // Create workflow run
    const wfRun = this.store.createWorkflowRun({
      issueNumber: issue.number,
      issueUrl: issue.url,
      repo: this.repo,
    });

    // Create agent run
    const agentRun = this.store.createAgentRun({
      workflowRunId: wfRun.id,
      phase: "triage",
    });

    // Launch triage phase
    const result = this.launcher.launch({
      phase: "triage",
      repoPath: this.repoRoot,
      runId: wfRun.id,
      input: {
        number: issue.number,
        title: issue.title,
        body: issue.body,
        labels: issue.labels,
        comments: issue.comments,
      },
    });

    // Complete agent run
    this.store.completeAgentRun(agentRun.id, {
      status: result.exitCode === 0 ? "success" : "failed",
      outputPath: result.outputPath,
      eventsPath: result.eventsPath,
    });

    if (result.exitCode !== 0) {
      this.store.updateStatus(wfRun.id, "failed", "Triage phase failed");
      await this.github.addComment(
        this.repo, issueNumber,
        "⚠️ **Triage failed.** Escalating for human review."
      );
      await this.github.addLabels(this.repo, issueNumber, ["da:blocked"]);
      return this.store.getWorkflowRun(wfRun.id)!;
    }

    // Post triage summary
    const output = result.output as Record<string, unknown> | null;
    const summary = (output as any)?.summary ?? "Triage complete.";
    await this.github.addComment(this.repo, issueNumber, `## 🔍 Triage Summary\n\n${summary}`);

    // Update state
    this.store.updateStatus(wfRun.id, "triaged", "Triage phase complete");
    await this.github.addLabels(this.repo, issueNumber, ["da:triaged"]);
    await this.github.removeLabels(this.repo, issueNumber, ["da:ready"]);

    return this.store.getWorkflowRun(wfRun.id)!;
  }
}
```

```typescript
async plan(issueNumber: number): Promise<WorkflowRun> {
  const wfRun = this.store.getWorkflowRunByIssue(this.repo, issueNumber);
  if (!wfRun) throw new Error(`No workflow run found for issue #${issueNumber}`);
  if (wfRun.status !== "triaged") {
    throw new Error(`Cannot plan: current status is ${wfRun.status}, expected triaged`);
  }

  const issue = await this.github.fetchIssue(this.repo, issueNumber);

  // Get triage output from the last agent run
  const agentRun = this.store.createAgentRun({
    workflowRunId: wfRun.id,
    phase: "plan",
  });

  const result = this.launcher.launch({
    phase: "plan",
    repoPath: this.repoRoot,
    runId: wfRun.id,
    input: {
      number: issue.number,
      title: issue.title,
      body: issue.body,
      labels: issue.labels,
      comments: issue.comments,
    },
  });

  this.store.completeAgentRun(agentRun.id, {
    status: result.exitCode === 0 ? "success" : "failed",
    outputPath: result.outputPath,
    eventsPath: result.eventsPath,
  });

  if (result.exitCode !== 0) {
    this.store.updateStatus(wfRun.id, "failed", "Plan phase failed");
    await this.github.addComment(
      this.repo, issueNumber,
      "⚠️ **Planning failed.** Escalating for human review."
    );
    return this.store.getWorkflowRun(wfRun.id)!;
  }

  const output = result.output as Record<string, unknown> | null;
  const summary = (output as any)?.summary ?? "Plan complete.";
  await this.github.addComment(
    this.repo, issueNumber,
    `## 📋 Implementation Plan\n\n${summary}\n\n---\n_Reply with feedback or \`/approve\` to proceed._`
  );

  this.store.updateStatus(wfRun.id, "plan_draft", "Plan draft posted");
  this.store.updateWorkflowRun(wfRun.id, { currentPhase: "plan" });

  return this.store.getWorkflowRun(wfRun.id)!;
}
```

```typescript
async implement(issueNumber: number): Promise<WorkflowRun> {
  const wfRun = this.store.getWorkflowRunByIssue(this.repo, issueNumber);
  if (!wfRun) throw new Error(`No workflow run found for issue #${issueNumber}`);
  if (wfRun.status !== "plan_accepted") {
    throw new Error(`Cannot implement: status is ${wfRun.status}, expected plan_accepted`);
  }

  this.store.updateStatus(wfRun.id, "implementing", "Starting implementation");
  this.store.updateWorkflowRun(wfRun.id, { currentPhase: "implement" });

  // Setup worktree for implementation
  const workDir = this.setupWorktree(issueNumber, wfRun.id);

  const issue = await this.github.fetchIssue(this.repo, issueNumber);

  const agentRun = this.store.createAgentRun({
    workflowRunId: wfRun.id,
    phase: "implement",
  });

  const result = this.launcher.launch({
    phase: "implement",
    repoPath: workDir,
    runId: wfRun.id,
    input: {
      number: issue.number,
      title: issue.title,
      body: issue.body,
      labels: issue.labels,
      comments: issue.comments,
    },
  });

  this.store.completeAgentRun(agentRun.id, {
    status: result.exitCode === 0 ? "success" : "failed",
    outputPath: result.outputPath,
    eventsPath: result.eventsPath,
  });

  if (result.exitCode !== 0) {
    this.store.updateStatus(wfRun.id, "failed", "Implementation failed");
    await this.github.addComment(
      this.repo, issueNumber,
      "⚠️ **Implementation failed.** See agent logs for details."
    );
    return this.store.getWorkflowRun(wfRun.id)!;
  }

  return this.store.getWorkflowRun(wfRun.id)!;
}
```

```typescript
async review(issueNumber: number): Promise<WorkflowRun> {
  const wfRun = this.store.getWorkflowRunByIssue(this.repo, issueNumber);
  if (!wfRun) throw new Error(`No workflow run for issue #${issueNumber}`);
  if (wfRun.status !== "draft_pr_opened") {
    throw new Error(`Cannot review: status is ${wfRun.status}`);
  }

  const workDir = wfRun.worktreePath ?? this.repoRoot;

  const agentRun = this.store.createAgentRun({
    workflowRunId: wfRun.id,
    phase: "review",
  });

  const result = this.launcher.launch({
    phase: "review",
    repoPath: workDir,
    runId: wfRun.id,
    input: {
      issueNumber,
      prNumber: wfRun.prNumber,
      branch: wfRun.branch,
    },
  });

  this.store.completeAgentRun(agentRun.id, {
    status: result.exitCode === 0 ? "success" : "failed",
    outputPath: result.outputPath,
    eventsPath: result.eventsPath,
  });

  if (result.exitCode !== 0) {
    this.store.updateStatus(wfRun.id, "failed", "Review phase failed");
    return this.store.getWorkflowRun(wfRun.id)!;
  }

  const output = result.output as any;
  const verdict = output?.result?.verdict ?? "pass";
  const blockingCount = output?.result?.blockingCount ?? 0;
  const summary = output?.summary ?? "Review complete.";

  await this.github.addComment(
    this.repo, issueNumber,
    `## 🔍 Auto Review\n\n${summary}`
  );

  if (verdict === "block" || blockingCount > 0) {
    this.store.updateStatus(wfRun.id, "auto_review_fix_loop", `Review found ${blockingCount} blocking issues`);
    this.store.updateWorkflowRun(wfRun.id, { currentPhase: "review" });
  } else {
    this.store.updateStatus(wfRun.id, "awaiting_human_review", "Auto review passed");
    await this.github.addLabels(this.repo, issueNumber, ["da:awaiting-human"]);
    await this.github.addComment(
      this.repo, issueNumber,
      "✅ **Auto review passed.** Ready for human review."
    );
  }

  return this.store.getWorkflowRun(wfRun.id)!;
}
```

```typescript
async repair(issueNumber: number): Promise<WorkflowRun> {
  const wfRun = this.store.getWorkflowRunByIssue(this.repo, issueNumber);
  if (!wfRun) throw new Error(`No workflow run for issue #${issueNumber}`);
  if (wfRun.status !== "auto_review_fix_loop") {
    throw new Error(`Cannot repair: status is ${wfRun.status}`);
  }

  const maxRounds = this.config.repair.max_rounds;
  const currentRound = wfRun.repairRound + 1;

  if (currentRound > maxRounds) {
    this.store.updateStatus(wfRun.id, "escalated", `Exceeded max repair rounds (${maxRounds})`);
    await this.github.addComment(
      this.repo, issueNumber,
      `⚠️ **Escalated.** Repair loop exceeded ${maxRounds} rounds. Human intervention needed.`
    );
    await this.github.addLabels(this.repo, issueNumber, ["da:escalated"]);
    return this.store.getWorkflowRun(wfRun.id)!;
  }

  const workDir = wfRun.worktreePath ?? this.repoRoot;

  const agentRun = this.store.createAgentRun({
    workflowRunId: wfRun.id,
    phase: "repair",
  });

  const result = this.launcher.launch({
    phase: "repair",
    repoPath: workDir,
    runId: wfRun.id,
    input: {
      round: currentRound,
      issueNumber,
      prNumber: wfRun.prNumber,
    },
  });

  this.store.completeAgentRun(agentRun.id, {
    status: result.exitCode === 0 ? "success" : "failed",
    outputPath: result.outputPath,
    eventsPath: result.eventsPath,
  });

  this.store.updateWorkflowRun(wfRun.id, {
    repairRound: currentRound,
    currentPhase: "repair",
  });

  if (result.exitCode !== 0) {
    this.store.updateStatus(wfRun.id, "failed", `Repair round ${currentRound} failed`);
    return this.store.getWorkflowRun(wfRun.id)!;
  }

  const summary = (result.output as any)?.summary ?? `Repair round ${currentRound} complete.`;
  await this.github.addComment(
    this.repo, issueNumber,
    `## 🔧 Repair Round ${currentRound}\n\n${summary}`
  );

  // Transition back to draft_pr_opened for re-review
  this.store.updateStatus(wfRun.id, "draft_pr_opened", `Repair round ${currentRound} complete, ready for re-review`);

  return this.store.getWorkflowRun(wfRun.id)!;
}
```

**Step 2: Implement runWorkflow method**

```typescript
async runWorkflow(
  issueNumber: number,
  options: { autoApprove?: boolean } = {},
): Promise<WorkflowRun> {
  // Phase 1: Triage
  let run = await this.triage(issueNumber);
  if (run.status === "failed") return run;

  // Phase 2: Plan
  run = await this.plan(issueNumber);
  if (run.status === "failed") return run;

  // Auto-approve or wait
  if (options.autoApprove) {
    run = await this.approvePlan(issueNumber);
  } else {
    // Stop here and wait for human approval
    return run;
  }

  // Phase 3: Implement
  run = await this.implement(issueNumber);
  if (run.status === "failed") return run;

  // Phase 4: Verify
  run = await this.verify(issueNumber);

  // Phase 5: Open PR
  run = await this.openPR(issueNumber);
  if (run.status === "failed") return run;

  // Phase 6: Review + Repair loop
  const maxRounds = this.config.repair.max_rounds;
  for (let round = 0; round <= maxRounds; round++) {
    run = await this.review(issueNumber);

    if (run.status === "awaiting_human_review") {
      // Review passed — done
      return run;
    }

    if (run.status === "auto_review_fix_loop") {
      run = await this.repair(issueNumber);
      if (run.status === "failed" || run.status === "escalated") {
        return run;
      }
      // repair transitions back to draft_pr_opened → loop continues
    }
  }

  return this.store.getWorkflowRun(
    this.store.getWorkflowRunByIssue(this.repo, issueNumber)!.id
  )!;
}
```
